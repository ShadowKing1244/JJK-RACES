<script>
    function spinGacha() {
        const zeninTechniques = ['Ten Shadows Technique', 'Projection Sorcery'];
        const gojoTechniques = ['Limitless', 'Six Eyes'];
        const kamoTechniques = ['Blood Manipulation'];
        const inumakiTechniques = ['Cursed Speech'];

        const clans = ['Zenin Clan', 'Gojo Clan', 'Kamo Clan', 'Inumaki Clan'];
        const randomClanIndex = Math.floor(Math.random() * clans.length);
        const selectedClan = clans[randomClanIndex];

        let selectedTechnique;
        switch (selectedClan) {
            case 'Zenin Clan':
                selectedTechnique = zeninTechniques[Math.floor(Math.random() * zeninTechniques.length)];
                break;
            case 'Gojo Clan':
                selectedTechnique = gojoTechniques[Math.floor(Math.random() * gojoTechniques.length)];
                break;
            case 'Kamo Clan':
                selectedTechnique = kamoTechniques[Math.floor(Math.random() * kamoTechniques.length)];
                break;
            case 'Inumaki Clan':
                selectedTechnique = inumakiTechniques[Math.floor(Math.random() * inumakiTechniques.length)];
                break;
            default:
                selectedTechnique = 'Unknown Technique';
        }

        document.getElementById('result').innerText = `You got: ${selectedClan} (${selectedTechnique})`;
    }
</script>
